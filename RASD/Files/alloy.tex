# If an internship position is open, it will eventually be closed
fact NotAlwaysOpenInternship {
    all i: Internship | 
        always ( i.isOpen = True
                implies 
                    eventually i.isOpen = False )
}
# If an internship is open it has been open since the beginning
fact NowOpenPreviouslyOpenInternship {
    all i: Internship | 
        always ( i.isOpen = True
                    implies 
                        historically i.isOpen = True )
}
# If an internship is open now it has always been open before
fact NowOpenPreviouslyOpen {
    all i: Internship | 
        always (i.isOpen = True 
            implies
                before i.isOpen = True)
}

}
